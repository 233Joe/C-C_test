stat	显示文件详细时间

cal 	显示日历

whoami	显示当前用户

tac	逆置

date	显示时间

clear	清屏

pwd	显示当前路径

tree (文件名)	把目录以树状结构显示(yum install -y tree 	安装tree)

./change	将文件让内容改大写

touch	创建文件

mv (文件名) (目标地址)	剪切

more	显示文件内容, 只显示到屏幕可见范围的内容（不可上翻下翻）
less	显示文件内容, 只显示到屏幕可见范围的内容（可上翻下翻）

head	显示文件前几行
tail	显示文件后几行

ls	显示当前目录文件(不包括隐藏文件)
{  
  ls -l 简写(ll)  显示当前目录文件更详细信息
  ls -a	显示读取目录的文件包括隐藏文件
  ls -d	不进入目录显示目录本身
  ls -r	对目录反向排序
  ls -R	列出所有子目录的文件
}

cd (目录名称)	进入目录
{
  cd ..	回退上级路径
  cd ~	进入工作(登录默认)目录
  cd -	回退最近一次所处路径
}

mkdir	创建目录
{
  mkdir -p ?/?/?/   创建一个路径的目录
}

rmdir	删除空目录
{
  rm	默认删除普通文件
  rm -r	递归删除
  rm -f	强制删除
  rm -rf	递归强制删除
  rm -i	询问删除
}

cp	拷贝文件
{
  cp -r	递归拷贝
  cp -f	强制拷贝
  cp -rf	递归强制拷贝
  cp -i	询问拷贝
}

cat	显示文件内容
{
  cat -n	显示行号
}
