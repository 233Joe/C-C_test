ls	显示当前目录文件(不包括隐藏文件)
{  
  ls -l 简写(ll)  显示当前目录文件更详细信息
  ls -a	显示读取目录的文件包括隐藏文件
  ls -d	不进入目录显示目录本身
  ls -r	对目录反向排序
  ls -R	列出所有子目录的文件

}
cd (目录名称)	进入目录
{
  cd ..	回退上级路径
  cd ~	进入工作(登录默认)目录
  cd -	回退最近一次所处路径
}

pwd	显示当前路径

tree (文件名)	把目录以树状结构显示
 yum install -y tree 	安装tree

touch	创建文件

mkdir	创建目录
{
  mkdir -p ?/?/?/   创建一个路径的目录
}

stat	显示文件详细时间

whoami		显示当前用户

clear	清屏
