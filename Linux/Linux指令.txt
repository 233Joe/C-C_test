
lscpu	查看cpu信息
lsmem	查看内存信息
df -h	查看磁盘使用情况
who	查看当前在线用户

stat	显示文件详细时间

wc -l	显示行号

cal 	显示日历

sort (文件名)    对文本内容进行排序

uniq (文件名)   对相邻的文本行进行去重
 
alias (命令)       给命令起别名	

whoami	显示当前用户

tac	逆置

date	显示时间

clear	清屏

pwd	显示当前路径

tree (文件名)	把目录以树状结构显示(yum install -y tree 	安装tree)

./change	将文件让内容改大写

touch	创建文件

mv (文件名) (目标地址)	剪切

more	显示文件内容, 只显示到屏幕可见范围的内容（不可上翻下翻）
less	显示文件内容, 只显示到屏幕可见范围的内容（可上翻下翻）

head	显示文件前几行
tail	显示文件后几行

ls	显示当前目录文件(不包括隐藏文件)
{  
  ll -l 简写(ll)  显示当前目录文件更详细信息
  -a	显示读取目录的文件包括隐藏文件
  -d	不进入目录显示目录本身
  -r	对目录反向排序
  -R	列出所有子目录的文件
}

cd (目录名称)	进入目录
{
  cd ..	回退上级路径
  cd ~	进入工作(登录默认)目录
  cd -	回退最近一次所处路径
}

mkdir	创建目录
{
  -p ?/?/?/   创建一个路径的目录
}

rmdir	删除空目录
{
  rm	默认删除普通文件
  rm -r	递归删除
  rm -f	强制删除
  rm -rf	递归强制删除
  rm -i	询问删除
}

cp	拷贝文件
{
  -r	递归拷贝
  -f	强制拷贝
  -rf	递归强制拷贝
  -i	询问拷贝
}

cat	显示文件内容
{
  -n	显示行号
}

grep -?  关键字  (文件名)    (对文本关键字筛选)
{
  -i	忽略大小写筛选
  -n	输出行号
  -v	反向选择，没有匹配到的显示
}

压缩与解压
{
  zip (文件名)  目录地址	压缩文件
  zip -r (文件名) 目录地址 	递归压缩文件

  unzip (文件名)	解压压缩包
  unzip (文件名) -d (路径)   解压到对应路径
}

tar
{
  -c	创建一个压缩文件
  -x	解开一个压缩文件
  -z	zip压缩
  -f	后面紧跟文件名
  -v	解压/压缩 时显示文件名

tar -czf 文件名 目录名	打包并压缩
tar -xzf 文件名		解包并解压
}

查找
{ 
  find (路径) -name (文件名)	查找文件
  {

  }
  which (命令)  查找命令的使用路径
  whereis (关键字)  查找关键字文件
}

热键
{
  tab	自动补齐目录/查询命令
  ctrl + c   终止前台运行程序
  ctrl + r   搜索历史命令，左右选择
  crtl + d  退出当前用户
}
